import hashlib
def getByteHash(originData):
    sha1_obj = hashlib.sha1()
    sha1_obj.update(originData.encode("utf-8"))

    return sha1_obj.hexdigest()

def get_api_sign(post_data):
    str = "aee4c425dbb2288b80c71347cc37d04b&425d8d985da3022ce5fd670765b30257"
    str1 = str
    api_sign = ""
    flag = True  #第一个字典不添加"&"
    #原始参数_1准备
    for key, value in post_data.items():
        #根据逆向的结果构造， = &
        if flag:
            tmp = key + "=" + value
            flag = False
        else:
            tmp = "&" + key + "=" + value
        str1 += tmp
    tmp2 = getByteHash(str1)

    #原始参数_2准备  
    str2 = str + tmp2
    api_sign = getByteHash(str2)

    return "OAuth api_sign=" + api_sign
